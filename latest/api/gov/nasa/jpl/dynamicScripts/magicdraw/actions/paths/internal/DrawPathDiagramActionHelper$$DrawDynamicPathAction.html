case class DrawDynamicPathAction(finalizationAction: DynamicPathFinalizationAction, diagram: DiagramPresentationElement, ID: String, name: String, key: KeyStroke, largeIcon: Icon) extends EnhancedDrawPathAction with Product with Serializable
